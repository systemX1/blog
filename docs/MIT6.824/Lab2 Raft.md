遇到更大的term 无论是AppendEntries RPC还是RequestVote RPC都要转变为Follower



```go
for {
    select {
        case <-heartbeat:
        rf.mu.Lock()
        if rf.stat != Leader {
            rf.mu.Unlock()
            break
        }
        rf.mu.Unlock()
        rf.appendEntries(nil)
        case t1 := <-rf.electTimer.C:
        rf.mu.Lock()
        if rf.stat == Leader {
            rf.mu.Unlock()
            break
        }
        rf.mu.Unlock()
        rf.startElection()
    }
}
```
